第一章：Python基础（电商案例） ▼

1.1 变量与数据类型

在电商数据分析中，我们常用变量存储商品信息、价格、销量等数据。

# 定义商品信息
product_name = "无线蓝牙耳机"
price = 199.0
sales = 150

# 计算销售额
revenue = price * sales
print(f"商品: {product_name}")
print(f"单价: {price}元")
print(f"销量: {sales}件")
print(f"销售额: {revenue}元")

1.2 条件判断

筛选符合条件的商品，如单价≥100元的商品。

products = [
    {"name": "手机", "price": 2999},
    {"name": "耳机", "price": 99},
    {"name": "电脑", "price": 5999}
]

for p in products:
    if p["price"] >= 100:
        print(f"{p['name']}: {p['price']}元")

1.3 循环与统计

使用循环统计订单总金额。

orders = [128, 256, 89, 312, 178, 45, 234, 87, 156, 201]
total = 0

for amount in orders:
    total += amount

print(f"10个订单总金额: {total}元")

1.4 列表操作

列表存储周销量，计算平均值和最大值。

weekly_sales = [120, 145, 98, 167, 134, 89, 156]

avg_sales = sum(weekly_sales) / len(weekly_sales)
max_sales = max(weekly_sales)

print(f"周平均销量: {avg_sales:.1f}件")
print(f"周最高销量: {max_sales}件")

1.5 字典与函数

字典存储用户信息，函数计算折扣价格。

# 字典存储用户信息
user = {
    "id": "U001",
    "name": "张三",
    "spent": 1280
}

print(f"用户ID: {user['id']}")
print(f"姓名: {user['name']}")
print(f"消费金额: {user['spent']}元")

# 满减函数
def calculate_price(price):
    if price >= 200:
        return price - 30
    return price

print(f"250元折扣后: {calculate_price(250)}元")
print(f"150元折扣后: {calculate_price(150)}元")

2.2 数据筛选与排序

筛选数据并按销量排序获取Top5商品。

import pandas as pd

data = {
    "商品名称": ["手机", "耳机", "电脑", "键盘", "鼠标", "充电宝", "数据线", "支架", "膜", "壳"],

df = pd.DataFrame(data)
top5 = df.sort_values("销量", ascending=False).head(5)

2.3 数据聚合与统计

按品类分组统计销售额。

import pandas as pd

data = {
    "商品名称": ["手机", "耳机", "电脑", "键盘", "鼠标"],
    "单价": [2999, 199, 5999, 129, 89],
    "销量": [150, 320, 80, 450, 520],
    "品类": ["数码", "配件", "数码", "配件", "配件"]
}

df = pd.DataFrame(data)
df["销售额"] = df["单价"] * df["销量"]

category_sales = df.groupby("品类")["销售额"].sum()
print("各品类总销售额:")
print(category_sales)

2.4 日期处理与用户分析

处理日期数据，筛选活跃用户。

import pandas as pd
from datetime import datetime, timedelta

data = {
    "用户ID": ["U001", "U002", "U003", "U004", "U005"],
    "最后活跃": ["2026-06-07", "2026-06-01", "2026-06-05", "2026-05-20", "2026-06-08"],
    "消费金额": [1280, 560, 2100, 890, 3200]
}

df = pd.DataFrame(data)
df["最后活跃"] = pd.to_datetime(df["最后活跃"])

seven_days_ago = datetime.now() - timedelta(days=7)
active_users = df[df["最后活跃"] >= seven_days_ago]
print("近7天活跃用户:")
print(active_users)

2.5 业务指标计算

计算客单价和订单转化率。

import pandas as pd

# 订单数据
orders_data = {
    "订单ID": ["O001", "O002", "O003", "O004", "O005"],
    "用户ID": ["U001", "U002", "U001", "U003", "U004"],
    "金额": [299, 159, 899, 45, 199]
}

orders_df = pd.DataFrame(orders_data)

# 计算客单价
total_revenue = orders_df["金额"].sum()
unique_users = orders_df["用户ID"].nunique()
avg_order_value = total_revenue / unique_users

# 模拟转化率计算
visitors = 500
buyers = unique_users
conversion_rate = (buyers / visitors) * 100

print(f"总销售额: {total_revenue}元")
print(f"购买用户数: {unique_users}人")
print(f"客单价: {avg_order_value:.2f}元")
print(f"访客数: {visitors}人")
print(f"订单转化率: {conversion_rate:.2f}%")
